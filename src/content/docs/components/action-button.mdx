## Usage

```tsx
import { ActionButton } from "@/components/ui/action-button"
```

```tsx
<ActionButton
  action={async () => {
    await new Promise((resolve) => setTimeout(resolve, 2000))
    alert("Action completed!")
  }}
>
  Click me
</ActionButton>
```
